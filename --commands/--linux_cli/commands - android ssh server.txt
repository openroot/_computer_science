3. systemctl enable ssh.service // To instead permanently enable the SSH service to start whenever the system is booted use

1. apt install samba
2. systemctl status smbd // or ; systemctl status smbd.service
3. systemctl --enable now smbd // or ; systemctl start smbd.service
4. ufw allow samba
5. usermod -aG sambashare devop

8. done ; test it local from android say vlc // smb://server-ip/shared-folder-path
